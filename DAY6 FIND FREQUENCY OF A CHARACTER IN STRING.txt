#include <stdio.h>
int countFrequency(char* str, char ch) {
    int count = 0;
    for (int i = 0; str[i] != '\0'; i++) {
        if (str[i] == ch) {
            count++;
        }
    }
    return count;
}
int main() {
    char str[] = "hello samantha";
    char ch = 'n';
    int frequency = countFrequency(str, ch);
    printf("Character '%c' appears %d times.\n", ch, frequency);
    return 0;
}
